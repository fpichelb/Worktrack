    private bool Loading = true;
    private User? User;
    private List<UserEntryViewModel> Entries = new();

    protected override async Task OnInitializedAsync()
    {
        // SecretCode aus URL laden
        var uri = new Uri(Nav.Uri);
        var query = System.Web.HttpUtility.ParseQueryString(uri.Query);
        var code = query.Get("code");

        if (string.IsNullOrWhiteSpace(code))
        {
            Loading = false;
            return;
        }

        // Benutzer laden
        User = await Db.Users
            .FirstOrDefaultAsync(u => u.SecretCode == code);

        if (User == null)
        {
            Loading = false;
            return;
        }

        // Eintrðge des Users laden
        var userEntries = await TimeEntries.GetEntriesForUserAsync(User.Id);

        // Eventnamen zuordnen
        Entries = userEntries
            .OrderByDescending(e => e.CheckIn)
            .Select(e => new UserEntryViewModel
            {
                Id = e.Id,
                EventName = Db.Events.First(ev => ev.Id == e.EventId).Name,
                Task = e.Task,
                CheckIn = e.CheckIn,
                CheckOut = e.CheckOut,
                DurationHours = e.DurationHours,
                Status = e.Status
            })
            .ToList();

        Loading = false;
    }

    private void GoLogin() => Nav.NavigateTo("/user/login", true);

    private void GoDashboard() =>
        Nav.NavigateTo($"/user/dashboard?code={User?.SecretCode}", true);

    {
